```javascript
public class Main{
  public static void main () {
                          //  a   |    b   |    c   |    x   |
      float a , b ;       //      |        |        |        | 
      int c , x ;         //      |        |        |        | 
      a = 6.0f;           //      |        |        |        | 
      b = a / 2;          //      |        |        |        | 
      c = 11;             //      |        |        |        | 
      x = (int) c / 4;    //      |        |        |        | 
      c = c % 2;          //      |        |        |        |

  }
}
``` 
